 16    def __init__(self, language: Language, **kwargs):
 17        """An agent that uses a language to communicate, e.g. a RSA pragmatic agent or a Lewis-Skyrms signaler.
 18
 19        Args:
 20            language: a language to construct a agent to define the relation between meanings and expressions, which can be used to initialize the agent matrices (e.g. `S` or `R`).
 21
 22            name: an optional string to name the communicative agent
 23        """
 24        self.language = language
 25
 26        # weight matrix indexing lookups
 27        self._referent_to_index = {
 28            referent: i for i, referent in enumerate(self.language.universe.referents)
 29        }
 30        self._index_to_referent = tuple(self.language.universe.referents)
 31        self._expression_to_index = {
 32            expression: i for i, expression in enumerate(self.language.expressions)
 33        }
 34        self._index_to_expression = tuple(self.language.expressions)
 35
 36        self.shape = None
 37        self.weights = None
 38
 39        if "name" in kwargs:
 40            self.name = kwargs["name"]
 41
 42        if "weights" in kwargs:
 43            self.weights = kwargs["weights"]

 57    def initialize_weights(self, weights: np.ndarray = None, initial="ones") -> None:
 58        """Initialize the agent's weight matrix.
 59
 60        Args:
 61            weights: an np.ndarray representing the weights to initialize the agent with. By default None, and the agent's weights will be initialized uniformly.
 62
 63            initial: {'ones', 'random'} a str reprsenting the initialization method to use. If 'ones' (default), initialize the weight matrix with `np.ones`. If 'random', initalize the weight matrix from `np.random.uniform`.
 64        """
 65        if weights is not None:
 66            if weights.shape != self.shape:
 67                raise ValueError(
 68                    f"Inapropriate Sender weight matrix shape for language. Sender is of shape {self.shape} but received {weights.shape}."
 69                )
 70            self.weights = weights
 71        else:
 72            # initialize equally
 73            if initial == "ones":
 74                self.weights = np.ones(self.shape)
 75            # initialize from uniform distribution
 76            elif initial == "random":
 77                self.weights = np.random.uniform(
 78                    low=0.0,
 79                    high=1.0,
 80                    size=self.shape,
 81                )
 82            else:
 83                raise ValueError(
 84                    f"Inappropriate value received for argument `initial`. Possible values are {'ones', 'random'}, but received: {initial}. "
 85                )

109    def sample_policy(self, index: int) -> int:
110        """Sample a communicative policy by uniformly sampling from a row vector of the agent's weight matrix specified by the index.
111
112        Args:
113            index: the integer index representing a row of the weight matrix.
114
115        Returns:
116            the integer index of the agent's choice
117        """
118        choices = self.weights[index]
119        choices_normalized = choices / choices.sum()
120        return np.random.choice(
121            a=range(len(choices)),
122            p=choices_normalized,
123        )

125    def to_language(
126        self,
127        data: dict = {
128            "complexity": None,
129            "accuracy": None,
130        },
131        threshold: float = 0.1,
132    ) -> Language:
133        """Get a language from the agent, representing its current (possibly learned) communicative behavior.
134
135        This function uses:
136            1. the agent's weight matrix,
137            1. the set of expression forms, and
138            1. the set of referents
139
140        from the language the agent was initialized with to generate a new language accurately reflecting the new expression meanings, e.g. how the agent interprets expressions as meaning zero or more referents.
141
142        Args:
143            threshold: a float in [0,1] representing the cutoff for determining if a meaning (referent) can be communicated by a expression. Because weights are not initialized to 0, it is a good idea to set nonzero values as the threshold.
144
145        Returns:
146            a Language corresponding to the form-meaning mapping defined by the communicative agent's weights.
147        """
148        # Construct the same kind of language as initialized with
149        language_type = type(self.language)
150        expression_type = type(self.language.expressions[0])
151        meaning_type = type(self.language.expressions[0].meaning)
152
153        # get distribution over communicative policies from a weight matrix
154        policies = self.normalized_weights()
155
156        expressions = []
157        # loop over agent's vocabulary
158        for old_expression in self.language.expressions:
159            # get all meanings that the expression can communicate
160            referents = [
161                referent
162                for referent in self.language.universe.referents
163                if policies[
164                    self.policy_to_indices(
165                        policy={"referent": referent, "expression": old_expression}
166                    )
167                ]
168                > threshold  # if probability of referent is high enough
169            ]
170
171            meaning = meaning_type(referents, self.language.universe)
172            # construct the updated expression as a new form-meaning mapping
173            expression = expression_type(
174                form=old_expression.form,
175                meaning=meaning,
176            )
177            expressions.append(expression)
178
179        if "name" not in data:
180            data["name"] = self.name
181
182        return language_type(expressions, data=data)

191    def __init__(self, language: Language, **kwargs):
192        super().__init__(language, **kwargs)

194    @property
195    def S(self) -> np.ndarray:
196        return self.weights
197
198    @S.setter
199    def S(self, mat: np.ndarray) -> None:
200        self.weights = mat
201
202    def normalized_weights(self) -> np.ndarray:
203        """Get the normalized weights of a Speaker.
204
205        Each row vector represents a conditional probability distribution over expressions, P(e | m).
206        """
207        # The sum of p(e | intended m) must be exactly 0 or 1.
208        # We check for nans because sometimes a language cannot express a particular meaning at all, resulting in a row sum of 0.
209        np.seterr(divide="ignore", invalid="ignore")
210        return np.nan_to_num(self.S / self.S.sum(axis=1, keepdims=True))

213class Listener(CommunicativeAgent):
214    def __init__(self, language: Language, **kwargs):
215        super().__init__(language, **kwargs)
216
217    @property
218    def R(self) -> np.ndarray:
219        return self.weights
220
221    @R.setter
222    def R(self, mat: np.ndarray) -> None:
223        self.weights = mat
224
225    def normalized_weights(self) -> np.ndarray:
226        """Normalize the weights of a Listener so that each row vector for the heard expression e represents a conditional probability distribution over referents P(m | e)."""
227        # The sum of p(m | heard e) must be 1. We can safely divide each row by its sum because every expression has at least one meaning.
228        return self.R / self.R.sum(axis=1, keepdims=True)

234"""In the Rational Speech Act framework, communicative agents reason recursively about each other's literal and pragmatic interpretations of utterances. Concretely, each agent is modeled by a conditional distribution. The speaker is represented by the probability of choosing to use an utterance (expression) given an intended meaning, P(e|m). The listener is a mirror of the speaker; it is represented by the probability of guessing a meaning given that they heard an utterance (expression), P(m|e).
235"""
236
237
238class LiteralSpeaker(Speaker):
239    """A literal speaker chooses utterances without any reasoning about other agents. The literal speaker's conditional probability distribution P(e|m) is uniform over all expressions that can be used to communicate a particular meaning. This is in contrast to a pragmatic speaker, whose conditional distribution is not uniform in this way, but instead biased towards choosing expressions that are less likely to be misinterpreted by some listener."""
240
241    def __init__(self, language: Language, **kwargs):
242        super().__init__(language, **kwargs)
243        self.S = self.language.binary_matrix()
244        self.S = self.normalized_weights()

247class LiteralListener(Listener):
248    """A naive literal listener interprets utterances without any reasoning about other agents. Its conditional probability distribution P(m|e) for guessing meanings is uniform over all meanings that can be denoted by the particular expression heard. This is in contrast to a pragmatic listener, whose conditional distribution is biased to guess meanings that a pragmatic speaker most likely intended."""
249
250    def __init__(self, language: Language, **kwargs):
251        super().__init__(language, **kwargs)
252        self.R = self.language.binary_matrix().T
253        self.R = self.normalized_weights()

256class PragmaticSpeaker(Speaker):
257    """A pragmatic speaker chooses utterances based on how a listener would interpret them. A pragmatic speaker may be initialized with any kind of listener, e.g. literal or pragmatic -- meaning the recursive reasoning can be modeled up to arbitrary depth."""
258
259    def __init__(
260        self,
261        language: Language,
262        listener: Listener,
263        **kwargs,
264    ):
265        """Initialize the |M|-by-|E| matrix, S, corresponding to the pragmatic speaker's conditional probability distribution over expressions given meanings.
266
267        The pragmatic speaker chooses expressions to communicate their intended meaning according to:
268
269            P(e | m) \propto exp(temperature * Utility(e,m))
270
271        where
272
273            Utility(e , m) := log(P_Listener(m | e))
274
275        Args:
276            language: the language with |M| meanings and |E| expressions defining the size of S.
277
278            listener: a communicative agent storing a matrix R representing the conditional distribution over expressions given meanings.
279
280            temperature: a float \in [0,1], representing how `optimally rational' the pragmatic speaker is; 1.0 is chosen when no particular assumptions about rationality are made.
281        """
282        super().__init__(language, **kwargs)
283
284        temperature = 1.0
285        if "temperature" in kwargs:
286            temperature = kwargs["temperature"]
287
288        # Row vector \propto column vector of literal R
289        self.S = softmax(np.nan_to_num(np.log(listener.R.T)) * temperature, axis=1)
290

292class PragmaticListener(Listener):
293    """A pragmatic listener interprets utterances based on their expectations about a pragmatic speaker's decisions. A pragmatic listener may be initialized with any kind of speaker, e.g. literal or pragmatic -- meaning the recursive reasoning can be modeled up to arbitrary depth."""
294
295    def __init__(
296        self,
297        language: Language,
298        speaker: Speaker,
299        prior: np.ndarray,
300        **kwargs,
301    ):
302        """Initialize the |E|-by-|M| matrix, R, corresponding to the pragmatic listener's conditional probability distribution over meanings given expressions.
303
304        The pragmatic listener chooses meanings as their best guesses of the expression they heard according to:
305
306            P(m | e) \propto P_PragmaticSpeaker(e | m)
307
308        Args:
309            language: the language with |M| meanings and |E| expressions defining the size of R.
310
311            speaker: a communicative agent storing a matrix S representing the  conditional distribution over expressions given meanings.
312
313            prior: a diagonal matrix of size |M|-by-|M| representing the communicative need probabilities for meanings.
314        """
315        super().__init__(language, **kwargs)
316        # Row vector \propto column vector of pragmatic S
317
318        self.R = np.zeros_like(speaker.S.T)
319        for i in range(len(self.R)):
320            col = speaker.S[:, i]
321            self.R[i] = col @ prior / np.sum(col @ prior)

324class BayesianListener(Listener):
325    """A Bayesian reciever chooses an interpretation according to p(meaning | word), where
326
327    $P(m | w) = \\frac{P(M | W) \cdot P(M)} { P(W) }$
328
329    Furthermore, we sometimes require that each word w is deterministically interpreted as meaning $\hat{m}$ as follows:
330
331    $\hat{m}_{w}(u) = \sum_m p(m|w) \cdot m(u)$
332
333    See altk.effcomm.information for more details.
334    """
335
336    def __init__(self, speaker: Speaker, prior: np.ndarray, name: str = None):
337        weights = bayes(speaker.normalized_weights(), prior)
338        super().__init__(speaker.language, weights=weights, name=name)
